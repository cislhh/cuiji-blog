class StateManager {
  constructor() {
    this.state = {}
    this.subscribers = new Map()
  }

  setState(key, value) {
    this.state[key] = value
    this.notify(key, value)
  }

  subscribe(key, callback) {
    if (!this.subscribers.has(key)) {

}
